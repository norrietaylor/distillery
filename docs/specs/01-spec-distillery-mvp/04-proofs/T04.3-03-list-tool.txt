from distillery.mcp.server import _handle_list
from distillery.store.duckdb import DuckDBStore
from distillery.mcp._stub_embedding import StubEmbeddingProvider
from distillery.models import Entry, EntryType, EntrySource

async def test():
    provider = StubEmbeddingProvider(dimensions=4)
    store = DuckDBStore(db_path=':memory:', embedding_provider=provider)
    await store.initialize()
    e1 = Entry(content='Python async programming', entry_type=EntryType.SESSION,
               source=EntrySource.CLAUDE_CODE, author='alice', project='api',
               tags=['python', 'async'])
    e2 = Entry(content='Meeting about caching', entry_type=EntryType.MINUTES,
               source=EntrySource.MANUAL, author='bob', project='infra',
               tags=['cache'])
    e3 = Entry(content='Python docs bookmark', entry_type=EntryType.BOOKMARK,
               source=EntrySource.MANUAL, author='alice', project='api',
               tags=['python', 'docs'])
    await store.store(e1)
    await store.store(e2)
    await store.store(e3)

    # Test: list all with defaults
    r = await _handle_list(store, {})
    data = json.loads(r[0].text)
    assert data['count'] == 3
    assert data['limit'] == 20
    assert data['offset'] == 0
    assert 'entries' in data
    print(f'PASS: list all defaults: count={data[\"count\"]}, limit={data[\"limit\"]}, offset={data[\"offset\"]}')

    # Test: list with author filter
    r = await _handle_list(store, {'author': 'alice', 'limit': 5})
    data = json.loads(r[0].text)
    assert data['count'] == 2

    print(f'PASS: list filter author=alice: count={data[\"count\"]}')

    # Test: list with entry_type filter

    # Test: list with tags filter
    r = await _handle_list(store, {'tags': ['python']})
    data = json.loads(r[0].text)

    print(f'PASS: list filter tags=[python]: count={data[\"count\"]}')

    # Test: pagination with offset
